## Step 2: Set Up the ThemeProvider with Global Style ðŸŽ¨

Now, let's set the stage with the `ThemeProvider`, and integrate the `GlobalStyle` for a consistent look across your application:

```javascript
import { themeStore } from "@mrtryy/fancyuilibary";
import { ThemeProvider, createGlobalStyle } from "styled-components";
import { TTheme } from '@mrtryy/fancyuilibary';
import { scrollbar }  from '@mrtryy/fancyuilibary'; /

// Define your Global Style
const GlobalStyle = createGlobalStyle<{ theme: TTheme; }>`
  body {
    margin: 0;
    padding: 0;
    background: ${({ theme }) => theme.colors.primary[0]};
    color: ${({ theme }) => theme.colors.secondary[0]};
    font-family: DMSans, Helvetica, Sans-Serif;
  }
  ${scrollbar};
`;

// ThemeStoreProvider with integrated Global Style
interface IThemeStoreProvider {
  children?: React.ReactNode;
}

export default function ThemeStoreProvider({ children }: IThemeStoreProvider) {
  const theme = themeStore((state) => state.theme);
  return (
    <ThemeProvider theme={theme}>
      <GlobalStyle theme={theme} />
      {children}
    </ThemeProvider>
  );
}
```
